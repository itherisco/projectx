import zmq
import json
import time
import sys

# CONFIGURATION
GATEWAY_ADDR = "tcp://127.0.0.1:5555"
AGENT_ID = "HUMAN_SOVEREIGN"

def start_terminal():
    context = zmq.Context()
    socket = context.socket(zmq.DEALER)
    socket.setsockopt(zmq.IDENTITY, AGENT_ID.encode())
    socket.connect(GATEWAY_ADDR)

    print("\n" + "="*50)
    print("       ITHERIS: SOVEREIGN COMMAND INTERFACE")
    print("       STATUS: AUTHENTICATED [ROOT OVERRIDE]")
    print("="*50)

    while True:
        try:
            # 1. Your input
            directive = input("\n[SOVEREIGN] > ").strip()
            if not directive: continue
            if directive.lower() in ["exit", "quit"]: break

            # 2. Package as a "Directive" (Always 100% confidence)
            packet = {
                "agent_id": AGENT_ID,
                "intent": "ACT", # Human commands are actions
                "payload": directive,
                "confidence": 1.0,
                "source_proof": "MANUAL_BYPASS_KEY_2026",
                "timestamp": int(time.time())
            }

            # 3. Send to the Rust Hub
            socket.send_json(packet)

            # 4. Wait for verified response
            if socket.poll(10000): # 10s timeout
                resp = socket.recv_json()
                print(f"\n[ITHERIS] > {resp.get('payload', 'System acknowledged.')}")
                print(f"[LAW VERDICT] > {resp.get('status', 'VERIFIED')}")
            else:
                print("\n[!] ALERT: Rust Hub is unresponsive.")

        except KeyboardInterrupt:
            break
    
    print("\nSovereign Terminal Session Closed.")

if __name__ == "__main__":
    start_terminal()
