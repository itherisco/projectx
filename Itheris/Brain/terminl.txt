import zmq
import json
import time
import uuid

# CONFIGURATION
GATEWAY_ADDR = "tcp://127.0.0.1:5555"
AGENT_ID = "HUMAN_SOVEREIGN"

def create_packet(payload, intent="THINK"):
    """Formats the message for the Rust Law Hub"""
    return {
        "agent_id": AGENT_ID,
        "intent": intent,
        "payload": payload,
        "confidence": 1.0,  # Human input is always 100% confident
        "source_proof": "MANUAL_OVERRIDE_KEY_2026",
        "timestamp": int(time.time())
    }

def run_terminal():
    context = zmq.Context()
    # Using a DEALER socket to talk to the Rust ROUTER
    socket = context.socket(zmq.DEALER)
    socket.setsockopt(zmq.IDENTITY, AGENT_ID.encode())
    socket.connect(GATEWAY_ADDR)

    print(f"--- ITHERIS SOVEREIGN TERMINAL ---")
    print(f"Connected to Rust Gateway at {GATEWAY_ADDR}")
    print(f"Type 'exit' to quit. System is listening...")

    while True:
        # 1. GET HUMAN INPUT
        user_input = input("\nYOU > ")
        
        if user_input.lower() in ['exit', 'quit']:
            break

        # 2. WRAP IN LAWFUL PACKET
        # We label this as 'THINK' so Julia processes it as a command
        packet = create_packet(user_input, intent="THINK")
        
        # 3. SEND TO RUST HUB
        socket.send_json(packet)

        # 4. WAIT FOR RESPONSE (From Julia via Rust)
        print("ITHERIS is thinking...", end="\r")
        try:
            # Polling to avoid infinite hang if Rust is down
            if socket.poll(10000):  # Wait 10 seconds
                response = socket.recv_json()
                print(f"ITHERIS > {response.get('payload', 'No payload received')}")
                if "status" in response:
                    print(f"[Law Verdict: {response['status']}]")
            else:
                print("Error: Rust Gateway timed out.")
        except Exception as e:
            print(f"\nCommunication Error: {e}")

if __name__ == "__main__":
    run_terminal()
