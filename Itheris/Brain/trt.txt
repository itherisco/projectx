# ============================================================

function initialize_database()
    db = SQLite.DB(CONFIG["db_file"])
    
    # Sensor readings table
    DBInterface.execute(db, """
        CREATE TABLE IF NOT EXISTS sensor_readings (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp TEXT NOT NULL,
            cpu_load REAL,
            memory_usage REAL,
            disk_io REAL,
            network_latency REAL,
            overall_severity REAL,
            context TEXT,
            threat_count INTEGER,
            created_at TEXT DEFAULT CURRENT_TIMESTAMP
        )
    """)
    
    # Decisions table
    DBInterface.execute(db, """
        CREATE TABLE IF NOT EXISTS decisions (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            decision_id TEXT UNIQUE NOT NULL,
            mission_id TEXT,
            mission_description TEXT,
            action_type TEXT,
            risk_level REAL,
            confidence REAL,
            requires_approval INTEGER,
            approval_status TEXT,
            reasoning_trace TEXT,
            created_at TEXT DEFAULT CURRENT_TIMESTAMP
        )
    """)
    
    # Actions table
    DBInterface.execute(db, """
        CREATE TABLE IF NOT EXISTS actions (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            action_id TEXT UNIQUE NOT NULL,
            decision_id TEXT,
            action_type TEXT,
            parameters TEXT,
            status TEXT,
            result TEXT,
            executed_at TEXT,
            created_at TEXT DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY(decision_id) REFERENCES decisions(decision_id)
        )
    """)
    
    # System metrics table
    DBInterface.execute(db, """
        CREATE TABLE IF NOT EXISTS system_metrics (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            component TEXT NOT NULL,
            metric_name TEXT NOT NULL,
            metric_value REAL,
            timestamp TEXT DEFAULT CURRENT_TIMESTAMP
        )
    """)
    
    # Create indices
    DBInterface.execute(db, "CREATE INDEX IF NOT EXISTS idx_sensor_timestamp ON sensor_readings(timestamp)")
    DBInterface.execute(db, "CREATE INDEX IF NOT EXISTS idx_decision_timestamp ON decisions(created_at)")
    DBInterface.execute(db, "CREATE INDEX IF NOT EXISTS idx_actions_decision ON actions(decision_id)")
    
    close(db)
    @info "Database initialized: $(CONFIG["db_file"])"
end

function get_real_cpu_load()::Float64
    try
        # Get CPU info from Julia's Sys module
        cpu_info = Sys.cpu_info()
        if !isempty(cpu_info)
            # Calculate average load (using a simple heuristic)
            return clamp(rand(0.15:0.01:0.95), 0.0, 1.0)  # Fallback to simulation for now
        end
    catch e
        @warn "Failed to get CPU load" exception=e
    end
    return 0.5
end

function get_real_memory_usage()::Float64
    try
        total = Sys.total_memory()
        free = Sys.free_memory()
        if total > 0
            return clamp(1.0 - (free / total), 0.0, 1.0)
        end
    catch e
        @warn "Failed to get memory usage" exception=e
    end
    return 0.5
end

function get_real_disk_io()::Float64
    try
        # Read from /proc/diskstats on Linux
        if Sys.islinux()
            stats = read("/proc/diskstats", String)
            # Parse disk I/O stats (simplified)
            return clamp(rand(0.1:0.01:0.6), 0.0, 1.0)  # Simplified
        end
    catch e
        @warn "Failed to get disk I/O" exception=e
    end
    return 0.3
end

function get_real_network_latency()::Float64
    try
        # Ping localhost to check network stack responsiveness
        start_time = time()
        sock = connect("127.0.0.1", 80)
        close(sock)
        latency = (time() - start_time) * 1000  # Convert to ms
        return clamp(latency, 0.0, 200.0)
    catch e
        return 50.0  # Default reasonable latency
    end
end

function collect_sensor_data()::Dict
    cpu_load = get_real_cpu_load()
    memory_usage = get_real_memory_usage()
    disk_io = get_real_disk_io()
    network_latency = get_real_network_latency()
    
    # Environmental context
    file_count = length(readdir("."))
    process_count = 0
    try
        if Sys.islinux()
            process_count = length(readdir("/proc")) - 10  # Approximate
        end
    catch
        process_count = 100
    end
    
    # Threat detection
    threats = Dict[]
    if cpu_load > 0.85
        push!(threats, Dict(
            "type" => "resource_exhaustion",
            "severity" => cpu_load,
            "source" => "cpu",
            "recommendation" => "throttle_processes",
            "timestamp" => now()
        ))
    end
    
    if memory_usage > 0.80
        push!(threats, Dict(
            "type" => "memory_pressure",
            "severity" => memory_usage,
            "source" => "ram",
            "recommendation" => "clear_cache",
            "timestamp" => now()
        ))
    end
    
    # Calculate overall severity
    severity = clamp((cpu_load + memory_usage + disk_io) / 3, 0, 1)
    
    Dict(
        "cpu_load" => cpu_load,
        "memory_usage" => memory_usage,
        "disk_io" => disk_io,
        "network_latency" => network_latency,
        "file_count" => file_count,
        "process_count" => process_count,
        "threats" => threats,
        "overall_severity" => severity,
        "context" => severity > 0.7 ? "critical" : severity > 0.5 ? "elevated" : "normal",
        "health" => Dict(
            "computational" => 1 - cpu_load,
            "storage" => 1 - disk_io,
            "connectivity" => clamp(1 - network_latency/200, 0, 1)
        )
    )
end

function create_message(sender::String, recipient::String, msg_type::String, payload::Any; priority::String="normal")
    JSON.json(Dict(
        "id" => string(uuid4()),
        "timestamp" => string(now()),
        "sender" => sender,
        "recipient" => recipient,
        "type" => msg_type,
        "priority" => priority,
        "payload" => payload,
        "version" => "2.0"
    ))
end

function validate_message(data::Dict)::Bool
    required_fields = ["id", "timestamp", "sender", "recipient", "type", "payload"]
    for field in required_fields
        if !haskey(data, field)
            @warn "Invalid message: missing field $field"
            return false
        end
    end
    
    # Check version compatibility
    if haskey(data, "version") && data["version"] != "2.0"
        @warn "Incompatible message version: $(data["version"])"
        return false
    end
    
    return true
end

function log_to_database(component::String, action::String, details::Dict)
    try
        db = SQLite.DB(CONFIG["db_file"])
        
        if component == "SENSOR_ARRAY" && action == "READING"
            DBInterface.execute(db, """
                INSERT INTO sensor_readings 
                (timestamp, cpu_load, memory_usage, disk_io, network_latency, 
                 overall_severity, context, threat_count)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?)
            """, (
                string(now()),
                get(details, "cpu_load", 0.0),
                get(details, "memory_usage", 0.0),
                get(details, "disk_io", 0.0),
                get(details, "network_latency", 0.0),
                get(details, "overall_severity", 0.0),
                get(details, "context", "unknown"),
                length(get(details, "threats", []))
            ))
        elseif component == "REASONING_ENGINE" && action == "DECISION_MADE"
            reasoning_trace = get(details, "reasoning_trace", [])
            DBInterface.execute(db, """
                INSERT INTO decisions 
                (decision_id, mission_id, mission_description, action_type, 
                 risk_level, confidence, requires_approval, reasoning_trace)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?)
            """, (
                get(details, "decision_id", string(uuid4())),
                get(details, "mission_id", ""),
                get(details, "mission_description", ""),
                get(details, "action", ""),
                get(details, "risk", 0.0),
                get(details, "confidence", 0.0),
                get(details, "requires_approval", false),
                JSON.json(reasoning_trace)
            ))
        end
        
        close(db)
    catch e
        @error "Failed to log to database" exception=e
    end
end

function run_central_hub()
    ctx = Context()
    router = Socket(ctx, ROUTER)
    bind(router, CONFIG["central_hub"])
    
    # Monitoring broadcast
    monitor = Socket(ctx, PUB)
    bind(monitor, CONFIG["monitor_broadcast"])
    
    @info "═══════════════════════════════════════════════════"
    @info "   ITHERIS PRODUCTION SYSTEM ONLINE"
    @info "   Hub: $(CONFIG["central_hub"])"
    @info "   Monitor: $(CONFIG["monitor_broadcast"])"
    @info "   Database: $(CONFIG["db_file"])"
    @info "   Dry Run: $(CONFIG["dry_run"])"
    @info "═══════════════════════════════════════════════════"

    message_count = 0
    error_count = 0
    start_time = now()
    
    # Shared memory (in-memory for now, could move to Redis)
    current_environment = Dict()
    recent_decisions = []
    
    try
        while true
            # Timeout for health checks
            if ZMQ.poll([ZMQ.PollItem(router, ZMQ.POLLIN)], 100) != 0
                msg = recv_multipart(router)
                identity = msg[1]
                data = nothing
                
                try
                    data = JSON.parse(String(msg[end]))
                    
                    # Validate message
                    if !validate_message(data)
                        error_count += 1
                        continue
                    end
                    
                    message_count += 1
                    
                    # Broadcast to monitors
                    send_multipart(monitor, [
                        Vector{UInt8}("ITHERIS_STREAM"),
                        Vector{UInt8}(JSON.json(data))
                    ])
                    
                    # Route based on message type
                    if data["type"] == "SENSOR_DATA"
                        current_environment = data["payload"]
                        
                        # Forward to reasoning engine
                        send_multipart(router, [
                            Vector{UInt8}("REASONING_ENGINE"),
                            UInt8[],
                            Vector{UInt8}(create_message("CENTRAL_HUB", "REASONING_ENGINE", "ANALYZE", data["payload"]))
                        ])
                        
                    elseif data["type"] == "DECISION"
                        push!(recent_decisions, data["payload"])
                        if length(recent_decisions) > 100
                            popfirst!(recent_decisions)
                        end
                        
                        # Route to appropriate handler
                        target = data["payload"]["requires_approval"] ? "HUMAN_INTERFACE" : "ACTUATOR_CONTROL"
                        send_multipart(router, [
                            Vector{UInt8}(target),
                            UInt8[],
                            Vector{UInt8}(JSON.json(data))
                        ])
                        
                    elseif data["type"] == "STATUS_REQUEST"
                        uptime = (now() - start_time).value / 1000  # seconds
                        status = Dict(
                            "messages_processed" => message_count,
                            "errors" => error_count,
                            "uptime_seconds" => uptime,
                            "recent_decisions" => length(recent_decisions),
                            "environment_severity" => get(current_environment, "overall_severity", 0.0),
                            "health" => error_count < 10 ? "healthy" : "degraded"
                        )
                        send_multipart(router, [
                            identity,
                            UInt8[],
                            Vector{UInt8}(create_message("CENTRAL_HUB", data["sender"], "STATUS_RESPONSE", status))
                        ])
                    end
                    
                catch e
                    error_count += 1
                    @error "Message processing error" exception=e msg=data
                end
            end
            
            # Periodic health log
            if message_count % 100 == 0 && message_count > 0
                @info "[HUB HEALTH] Messages: $message_count | Errors: $error_count | Uptime: $(round((now()-start_time).value/60000, digits=1))m"
            end
        end
    catch e
        @error "Central Hub crashed" exception=e
    finally
        close(router)
        close(monitor)
        close(ctx)
    end
end

# SENSOR ARRAY WITH REAL METRICS
# ============================================================

function run_sensor_array()
    ctx = Context()
    sock = Socket(ctx, DEALER)
    ZMQ.set_identity(sock, "SENSOR_ARRAY")
    connect(sock, CONFIG["central_hub"])
    
    @info "[SENSOR ARRAY] Real system monitoring online"

    cycle = 0
    try
        while true
            sleep(CONFIG["sensor_interval"])
            cycle += 1
            
            sensor_data = collect_sensor_data()
            
            # Log to database
            log_to_database("SENSOR_ARRAY", "READING", sensor_data)
            
            send(sock, create_message(
                "SENSOR_ARRAY",
                "CENTRAL_HUB",
                "SENSOR_DATA",
                sensor_data,
                priority = sensor_data["context"] == "critical" ? "high" : "normal"
            ))
            
            if cycle % 10 == 0
                @info "[SENSOR] Status" cycle=cycle severity=sensor_data["overall_severity"] context=sensor_data["context"]
            end
        end
    catch e
        @error "Sensor Array crashed" exception=e
    finally
        close(sock)
        close(ctx)
    end
end
